import PropTypes from 'prop-types'
import { FontAwesomeIcon } from "@fortawesome/react-fontawesome";
import { faPen, faCircleMinus } from '@fortawesome/free-solid-svg-icons'
import './post-card.style.css'

/**
 * Post item card
 *
 * @component
 * @example
 * const item = {
 *   title: 'Post title',
 *   body: 'Post body'
 * }
 * const handleEdit = () => {}
 * const handleDelete = () => {}
 * return (
 *   <PostCard item={item} onEdit={handleEdit} onDelete={handleDelete} />
 * )
 */
export const PostCard = ({ item, onEdit, onDelete }) => {
  return (
    <div className="post-card">
      <h1 className="post-card__title">
        {item.title}
        <span className="post-card__actions">
          <FontAwesomeIcon icon={faPen} onClick={onEdit.bind(null, item)} color="blue" />
          <FontAwesomeIcon icon={faCircleMinus} onClick={onDelete.bind(null, item)} color="red" />
        </span>
      </h1>
      <p className="post-card__body">{item.body}</p>
    </div>
  )
}

PostCard.propTypes = {
  item: PropTypes.shape({
    title: PropTypes.string,
    body: PropTypes.string
  }).isRequired,
  onEdit: PropTypes.func.isRequired,
  onDelete: PropTypes.func.isRequired
}
